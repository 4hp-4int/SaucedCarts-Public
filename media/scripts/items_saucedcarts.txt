module SaucedCarts
{
    /* Deliberately no `imports {Base}` here.
     *
     * The recipe file declares `module Base { imports {SaucedCarts} }` so
     * the recipe deserializes cleanly server-side in MP (vanilla's
     * NetTimedAction.parse looks up the recipe by module.name and expects
     * `module Base`). If we ALSO imported Base here, the two files form a
     * cycle and PZ's ScriptBucket.get infinite-recurses → stack overflow.
     *
     * Item defs below reference Base items via the `base:` prefix
     * (e.g. `ItemType = base:container`, `Tags = ...;base:heavyitem`),
     * which PZ resolves independently of the `imports` clause. So the
     * import isn't actually needed here.
     */

    /* ==================== SHOPPING CART ==================== */
    /* Large metal cart for hauling supplies from stores */
    item ShoppingCart
    {
        DisplayCategory = Container,
        DisplayName = Shopping Cart,
        ItemType = base:container,
        Weight = 8.0,
        Icon = ShoppingCart,
        Capacity = 50,
        WeightReduction = 95,
        RunSpeedModifier = 0.75,
        RequiresEquippedBothHands = true,
        ConditionMax = 100,
        ConditionLowerChance = 25,
        CloseSound = LockerMetalClose,
        OpenSound = LockerMetalOpen,
        PutInSound = LockerMetalTransferItem,
        Tags = SaucedCart;base:heavyitem,
        Tooltip = Tooltip_SaucedCarts_ShoppingCart,
        TorchDot = 0.5,
        StaticModel = ShoppingCartModel,
        WorldStaticModel = ShoppingCartModel,
        ReplaceInPrimaryHand = ShoppingCartModel holdingcartright,
        ReplaceInSecondHand = ShoppingCartModel holdingcartleft,
    }
}
